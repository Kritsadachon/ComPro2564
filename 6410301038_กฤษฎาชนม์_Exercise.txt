print('enter number : ',end = " ")
number_input = int(input())

i = 1
while(i<=number_input):
    j = 1
    k=i
    while(j<=i):
        if(j==1 or j==k):
            print("*", end = "")

        else:
            print('' ,end = " ")

        j+=1

    print('')
    i+=1
